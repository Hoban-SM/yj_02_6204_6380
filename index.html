/** * 입주예약 시스템 - Google Apps Script 백엔드 (운영자 서버) * ------------------------------------------------------- * 연결된 스프레드시트 시트 구성 * 1) resident : id | dong | ho | name | phone_last4 * 2) reservation : id | resident_id | dong | ho | reservation_date | start_time | end_time | group_no | status | created_at * * 배포 방법 * 1. 확장 프로그램 > Apps Script 에서 이 파일 내용을 Code.gs 에 붙여넣기 * 2. 배포 > 새 배포 > 유형: 웹 앱 * - 실행 계정: 나 * - 액세스 권한: 모든 사용자 (Anyone) * 3. 발급된 웹앱 URL을 site/index.html 의 GAS_URL 에 넣기 * * CORS 메모 * - GitHub Pages(정적 사이트)에서 호출하므로 브라우저 preflight를 피하기 위해 * GET은 쿼리스트링, POST는 Content-Type: text/plain 으로 보냅니다. * (프론트엔드 site/index.html 에 동일하게 구현되어 있음) */ const SS = SpreadsheetApp.getActiveSpreadsheet(); const RESIDENT_SHEET = 'resident'; const RESERVATION_SHEET = 'reservation'; const PERIOD_START = '2026-06-29'; const PERIOD_END = '2026-08-28'; const TIME_SLOTS = [ ['09:00', '11:00'], ['11:00', '13:00'], ['13:00', '15:00'], ['15:00', '17:00'] ]; /* ================= 라우팅 ================= */ function doGet(e) { try { const action = e.parameter.action; if (action === 'login') return json(login(e.parameter)); if (action === 'availability') return json(getAvailability(e.parameter)); if (action === 'myReservation') return json(getMyReservation(e.parameter)); return json({ ok: false, error: '알 수 없는 요청입니다.' }); } catch (err) { return json({ ok: false, error: String(err) }); } } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action; if (action === 'reserve') return json(reserve(body)); if (action === 'cancel') return json(cancel(body)); return json({ ok: false, error: '알 수 없는 요청입니다.' }); } catch (err) { return json({ ok: false, error: String(err) }); } } function json(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } /* ================= 공통 유틸 ================= */ function getSheet(name) { const sh = SS.getSheetByName(name); if (!sh) throw new Error('시트를 찾을 수 없습니다: ' + name); return sh; } function sheetToObjects(sheet) { const values = sheet.getDataRange().getValues(); const headers = values[0]; return values.slice(1) .filter(function (row) { return row.some(function (c) { return c !== ''; }); }) .map(function (row) { const obj = {}; headers.forEach(function (h, i) { obj[h] = row[i]; }); return obj; }); } // 호수 뒷자리 2자리 -> 그룹 판정 (1~10호: 그룹1, 11~14호: 그룹2) function getLineGroup(ho) { const digits = String(ho).replace(/[^0-9]/g, ''); const last2 = parseInt(digits.slice(-2), 10); if (isNaN(last2)) throw new Error('호수 형식이 올바르지 않습니다: ' + ho); return (last2 >= 1 && last2 <= 10) ? 1 : 2; } function normalizePhone(v) { return String(v).replace(/[^0-9]/g, ''); } function formatDate(d) { if (typeof d === 'string') return d.slice(0, 10); const dt = new Date(d); const y = dt.getFullYear(); const m = String(dt.getMonth() + 1).padStart(2, '0'); const day = String(dt.getDate()).padStart(2, '0'); return y + '-' + m + '-' + day; } // 시트가 "09:00" 같은 텍스트를 시간(Date) 값으로 자동 변환해 저장하는 경우가 있어, // Date 객체든 문자열이든 항상 'HH:mm' 형식의 문자열로 통일해서 반환합니다. function formatTime(v) { if (v instanceof Date) { return Utilities.formatDate(v, 'Asia/Seoul', 'HH:mm'); } return String(v).trim(); } function stripTime(d) { return new Date(d.getFullYear(), d.getMonth(), d.getDate()); } function isWithinPeriod(dateStr) { const d = stripTime(new Date(dateStr)); return d >= stripTime(new Date(PERIOD_START)) && d <= stripTime(new Date(PERIOD_END)); } /* ================= 로그인 ================= */ function login(p) { const dong = String(p.dong || '').trim(); const ho = String(p.ho || '').trim(); const name = String(p.name || '').trim(); const phone = normalizePhone(p.phone); if (!dong || !ho || !name || !phone) { return { ok: false, error: '동 / 호 / 성명 / 휴대폰 뒷자리를 모두 입력해 주세요.' }; } const residents = sheetToObjects(getSheet(RESIDENT_SHEET)); const found = residents.find(function (r) { return String(r.dong).trim() === dong && String(r.ho).trim() === ho && String(r.name).trim() === name && normalizePhone(r.phone_last4) === phone; }); if (!found) { return { ok: false, error: '입력하신 정보와 일치하는 계약자를 찾을 수 없습니다.' }; } return { ok: true, resident: { id: found.id, dong: found.dong, ho: found.ho, name: found.name, group_no: getLineGroup(found.ho) } }; } /* ================= 가용성 조회 ================= */ function getAvailability(p) { const yearMonth = p.month; // 'YYYY-MM' const groupNo = Number(p.group_no); const residentId = p.resident_id; if (!yearMonth) return { ok: false, error: '월 정보가 없습니다.' }; const reservations = sheetToObjects(getSheet(RESERVATION_SHEET)) .filter(function (r) { return r.status === 'CONFIRMED'; }); const dates = getDatesInMonthWithinPeriod(yearMonth); const days = dates.map(function (dateStr) { const slots = TIME_SLOTS.map(function (slot) { const start = slot[0], end = slot[1]; const conflict = reservations.find(function (r) { return formatDate(r.reservation_date) === dateStr && formatTime(r.start_time) === start && Number(r.group_no) === groupNo; }); const mine = conflict && String(conflict.resident_id) === String(residentId); return { start: start, end: end, status: conflict ? (mine ? 'MINE' : 'TAKEN') : 'AVAILABLE', reservation_id: conflict ? conflict.id : null }; }); return { date: dateStr, slots: slots }; }); return { ok: true, days: days }; } function getDatesInMonthWithinPeriod(yearMonth) { const parts = yearMonth.split('-').map(Number); const y = parts[0], m = parts[1]; const periodStart = stripTime(new Date(PERIOD_START)); const periodEnd = stripTime(new Date(PERIOD_END)); const result = []; const daysInMonth = new Date(y, m, 0).getDate(); for (let d = 1; d <= daysInMonth; d++) { const dt = new Date(y, m - 1, d); if (dt >= periodStart && dt <= periodEnd) { result.push(formatDate(dt)); } } return result; } /* ================= 내 예약 조회 ================= */ function getMyReservation(p) { const residentId = p.resident_id; const reservations = sheetToObjects(getSheet(RESERVATION_SHEET)) .filter(function (r) { return r.status === 'CONFIRMED' && String(r.resident_id) === String(residentId); }); if (reservations.length === 0) return { ok: true, reservation: null }; const r = reservations[0]; return { ok: true, reservation: { id: r.id, date: formatDate(r.reservation_date), start_time: formatTime(r.start_time), end_time: formatTime(r.end_time) } }; } /* ================= 예약 생성 ================= */ function reserve(p) { const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const residentId = p.resident_id; const ho = p.ho; const date = p.date; const start = p.start_time; const end = p.end_time; const groupNo = getLineGroup(ho); if (!isWithinPeriod(date)) { return { ok: false, error: '예약 가능 기간(2026-06-29 ~ 2026-08-28)이 아닙니다.' }; } const validSlot = TIME_SLOTS.some(function (s) { return s[0] === start && s[1] === end; }); if (!validSlot) { return { ok: false, error: '올바르지 않은 시간대입니다.' }; } const sheet = getSheet(RESERVATION_SHEET); const reservations = sheetToObjects(sheet).filter(function (r) { return r.status === 'CONFIRMED'; }); // 세대당 1건 운영: 본인 기존 예약이 있으면 신규 예약 차단 const already = reservations.find(function (r) { return String(r.resident_id) === String(residentId); }); if (already) { return { ok: false, error: '이미 예약된 일정이 있습니다. 기존 예약을 취소한 뒤 다시 예약해 주세요.' }; } // 같은 날짜 + 같은 시간대 + 같은 그룹 충돌 체크 const conflict = reservations.find(function (r) { return formatDate(r.reservation_date) === date && formatTime(r.start_time) === start && Number(r.group_no) === groupNo; }); if (conflict) { return { ok: false, error: '이미 예약이 마감된 시간대입니다.' }; } const id = 'R' + new Date().getTime(); sheet.appendRow([id, residentId, p.dong, ho, date, start, end, groupNo, 'CONFIRMED', new Date()]); return { ok: true, reservation: { id: id, date: date, start_time: start, end_time: end } }; } finally { lock.releaseLock(); } } /* ================= 예약 취소 ================= */ function cancel(p) { const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const sheet = getSheet(RESERVATION_SHEET); const values = sheet.getDataRange().getValues(); const headers = values[0]; const idCol = headers.indexOf('id'); const residentCol = headers.indexOf('resident_id'); const statusCol = headers.indexOf('status'); for (let i = 1; i < values.length; i++) { if (String(values[i][idCol]) === String(p.reservation_id) && String(values[i][residentCol]) === String(p.resident_id)) { sheet.getRange(i + 1, statusCol + 1).setValue('CANCELED'); return { ok: true }; } } return { ok: false, error: '예약 내역을 찾을 수 없습니다.' }; } finally { lock.releaseLock(); } }
